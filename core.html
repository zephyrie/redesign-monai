Example Usage

from monai.transforms import ( Compose, LoadImage, ScaleIntensity, AddChannel ) # Define transforms for image preprocessing transforms = Compose([ LoadImage(image_only=True), AddChannel(), ScaleIntensity() ]) # Apply transforms to your image image = transforms(image_path)
